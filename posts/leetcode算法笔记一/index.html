import java.util.HashMap;
import java.util.Map;
class Solution {
    // 将差存入hashmap
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i< nums.length; i++){
            if (map.containsKey(nums[i])){
                return new int[]{map.get(nums[i]), i};
            }
            map.put(target - nums[i], i);
        }
        return null;
    }
    public int[] twoSum(int[] nums, int target) {
        for (int i = 0; i< nums.length-1; i++){
            for (int j = i+1; j < nums.length; j++){
                if (nums[i] + nums[j] == target){
                    return new int[]{i, j};
                }
            }
        }
        return null;
    }

}

/**
 * Definition for singly-linked list.
 * public class ListNode {
 * int val;
 * ListNode next;
 * ListNode() {}
 * ListNode(int val) { this.val = val; }
 * ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode head = new ListNode();
        sumOfTwoNumbers(l1, l2, head, 0);
        return head.next;
    }

    public void sumOfTwoNumbers(ListNode l1, ListNode l2, ListNode head, int gap) {
        if (l1 == null && l2 == null && gap == 0){
            return;
        }else if (l1 == null && l2 == null && gap != 0){
            head.next = new ListNode(gap);
            return;
        } else if (l1 == null){
            int sum = l2.val + gap;
            head.next = new ListNode(sum % 10);
            sumOfTwoNumbers(null, l2.next, head.next, sum / 10);
        }else if (l2 == null){
            int sum = l1.val + gap;
            head.next = new ListNode(sum % 10);
            sumOfTwoNumbers(l1.next, null, head.next, sum / 10);
        }else {
            int sum = l1.val + l2.val + gap;
            head.next = new ListNode(sum%10);
            sumOfTwoNumbers(l1.next, l2.next, head.next, sum / 10);
        }
    }
}

给定一个字符串 s ，请你找出其中不含有重复字符的 最长子串 的长度。 

示例 1: 
输入: s = "abcabcbb"
输出: 3 
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。

示例 2: 
输入: s = "bbbbb"
输出: 1
解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。

输入: s = "pwwkew"
输出: 3

示例 4: 
输入: s = ""
输出: 0

提示： 
0 <= s.length <= 5 * 10⁴ 
s 由英文字母、数字、符号和空格组成

// 方法一
public int lengthOfLongestSubstring(String s) {
  int ret = 0;
  // 定义一个窗口
  Set<Character> window = new HashSet<>();
  int len = s.length();
  // 窗口左指针
  int left = 0;
  // 窗口右指针
  int right = 0;
  while (left < len){
    while (right < len){
      boolean add = window.add(s.charAt(right));
      ret = Math.max(ret, window.size());
      if (!add){
        // 如果当前字符已经在窗口中，则去除左指针指向的字符。
        window.remove(s.charAt(left));
        break;
      }
      // 当前字符成功加到窗口函数，右指针右移一位。
      right++;
    }
    // 左指针右移一位。
    left++;
  }
  return ret;
}

// 方法二，这个方法最牛的地方就是用128个int，存储所有的字符
public int lengthOfLongestSubstring(String s) {
  int ret = 0;
  // 将字符以ASCII保存，记录最后出现的位置
  int[] last = new int[128];
  // 记录初始索引
  int start = 0;
  for (int i = 0; i < s.length(); i++){
    // 当前字符上一次出现位置的下一个字符为开始索引
    start = Math.max(start, last[s.charAt(i)]);
    // 获取最大字符长度
    ret = Math.max(ret, i - start + 1);
    // 记录当前字符当前位置
    last[s.charAt(i)] = i + 1;
  }
  return ret;
}

// 做法一: 将两个数组合并为一个数组，然后求取中位数
// 做法二: 利用二分的思想
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        double ret = 0;
        int len1 = nums1.length;
        int len2 = nums2.length;
        int left = (len1 + len2 + 1) / 2;
        int right = (len1 + len2 + 2) / 2;
        ret = (getMid(nums1, nums2, 0, 0, left) + getMid(nums1, nums2, 0, 0, right)) / 2.0;
        return ret;
    }

    public int getMid(int[] nums1, int[] nums2, int index1, int index2, int k) {
        if (index1 >= nums1.length) {
            return nums2[index2 + k - 1];
        }
        if (index1 >= nums2.length) {
            return nums1[index2 + k - 1];
        }
        if (k==1){
            return Math.min(nums1[index1], nums2[index2]);
        }
        int midVal1 = index1 + k/2 - 1 < nums1.length? nums1[index1 + k/2 - 1]: Integer.MAX_VALUE;
        int midVal2 = index2 + k/2 - 1 < nums2.length? nums2[index2 + k/2 - 1]: Integer.MAX_VALUE;
        if(midVal1 < midVal2){
            return getMid(nums1, nums2,index1 + k / 2, index2 , k - k / 2);
        }else{
            return getMid(nums1, nums2, index1 , index2 + k / 2,k - k / 2);
        }
    }

/**
*动态规划
*ababa是否是回文，要看bab是否是回文
*遍历时，外层循环以子串长度遍历
*/
public String longestPalindrome(String s){
    boolean target [][] = new boolean[s.length()][s.length()];
    for (int i = 0; i < s.length(); i++){
        target[i][i] = true;
    }
    int begin = 0;
  	int maxLength = 1;
  	for (int l = 2; j <= s.length(); i++){
      	for (int i = 0; i<s.length(); i++){
          	int j = i + l - 1;
          	if (j>=s.length()){
              	break;
            }
          	if (s.charAt(i) != s.charAt(j)){
              	target[i][j] = false;
            }else {
              	if (l<=2){
                  	target[i][j] = true;
                }else {
                  	target[i][j] = target[i+1][j-1];
                }
            }
          if (target[i][j] && l > maxLength){
            	begin = i;
            	maxLength = l;
          }
        }
    }
  	return s.substring(begin, begin+maxLength);
}
